tensor = np.zeros([3,4,5])

for i in [0]: #slice
    for j in [0,1]: #rows
        for k in [1,2,3]: # columns
            tensor[i,j,k] = 1
            
for i in [1]:
    for j in [0,1,2,3]:
        for k in [1,2,3]:
            tensor[i,j,k] = 1
 
for i in [2]:
    for j in [0,1]:
        for k in [0,1,2,3,4]:
            tensor[i,j,k] = 1
            
tensor = 2*np.array(tensor,dtype=np.int8)-1
